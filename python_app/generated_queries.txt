PREFIX lind: <http://solid-pod-1/ontology#>
PREFIX conc: <http://solid-pod-2/ontology#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT DISTINCT ?component ?componentName
WHERE {
  SERVICE <http://localhost:3030/solid-pod-1/sparql> {
    ?component a lind:Component ;
               lind:productFamily "NORTEC" ;
               rdfs:label ?componentName .
  }
  SERVICE <http://localhost:3030/solid-pod-2/sparql> {
    ?inventoryItem a conc:InventoryItem ;
                   conc:componentType ?component ;
                   conc:isAvailableForReuse true .
  }
}
